# === GIT REMOTE -V COMMAND GUIDE ===

# What git remote -v does:
# Shows all remote repositories linked to your local project
# -v flag shows verbose output (URLs for fetch and push)

git remote -v

# Expected output should look like:
# origin  https://github.com/daiyyanholmes88/Myfirstproject.git (fetch)
# origin  https://github.com/daiyyanholmes88/Myfirstproject.git (push)

# === COMMON REMOTE COMMANDS ===

# Check current remotes
git remote -v

# Add new remote
git remote add origin https://github.com/daiyyanholmes88/Myfirstproject.git

# Change remote URL
git remote set-url origin https://github.com/daiyyanholmes88/Myfirstproject.git

# Remove remote
git remote remove origin

# Rename remote
git remote rename origin upstream

# === TROUBLESHOOTING ===

# If no output or wrong URL:
# 1. Check if you're in the right project folder
# 2. Make sure git init was run
# 3. Add the correct remote URL

# If you see wrong repository URL:
git remote set-url origin https://github.com/daiyyanholmes88/Myfirstproject.git

# Test connection to remote
git ls-remote origin

# === USAGE EXAMPLES ===

# Before pushing, always check your remote:
git remote -v
git push origin main

# When cloning, remote is automatically set:
git clone https://github.com/daiyyanholmes88/Myfirstproject.git
cd Myfirstproject
git remote -v
